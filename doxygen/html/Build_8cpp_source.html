500 void CBuild::Toolchain::call(std::vector<std::string> *args, bool force,
501  bool debug, bool dummy) {
502  CBuild::print("Starting " + this->id + " toolchain in build mode ",
503  CBuild::color::RED);
504  // Save scratch variables
505  this->args = args;
506  this->force = force;
507  this->dummy = dummy;
508  // Special link tag for dynamick libraries
509  if (this->build_type == CBuild::DYNAMIC_LIBRARY) {
510  this->compiler_args.push_back("-fPIC");
511  this->link_args.push_back("-shared");
512  }
513  // Debug mode of compilation
514  if (debug)
515  this->compiler_args.push_back("-g");
516  // Some standart compile and link args
517  this->add_compile_arg("-I" + CBUILD_CACHE_DIR + "/" +
518  CBUILD_PROJECT_DEPS_HEADERS);
519  this->add_link_arg("-I" + CBUILD_CACHE_DIR + "/" +
520  CBUILD_PROJECT_DEPS_HEADERS);
521  this->add_compile_arg("-L" + CBUILD_CACHE_DIR + "/" +
522  CBUILD_PROJECT_DEPS_DIR);
523  this->add_link_arg("-L" + CBUILD_CACHE_DIR + "/" + CBUILD_PROJECT_DEPS_DIR);
524  // Init dirs
525  this->init();
526  // For packing in deb fomat
527  if (this->build_type == CBuild::DYNAMIC_LIBRARY) {
528  std::string lib = this->gen_out_name();
529  int pos = lib.find_last_of('/');
530  lib = lib.substr(pos + 1);
531  if (this->version_major < 0) {
532  this->add_link_arg(" -Wl,-soname," + lib);
533  this->add_compile_arg(" -Wl,-soname," + lib);
534  } else {
535  this->add_link_arg(" -Wl,-soname," + lib + std::string(".") +
536  std::to_string(this->version_major));
537  this->add_compile_arg(" -Wl,-soname," + lib + std::string(".") +
538  std::to_string(this->version_major));
539  }
540  }
541  // Init
542  CBuild::print("Calling tasks marked as PRE ", CBuild::color::GREEN);
543  // Call all dependencies (tasks) marked as PRE
544  for (unsigned int i = 0; i < this->required.size(); i++) {
545  auto elem = this->required.at(i);
546  if (elem.data == CBuild::PRE) {
547  CBuild::Registry::CallTask(elem.key, {});
548  }
549  }
550  CBuild::print("Calling all required toolchains ", CBuild::color::GREEN);
551  // Call all dependencies (toolchains)
552  for (std::string id : this->depends) {
553  // Call target
554  auto target = CBuild::Registry::GetToolchain(id);
555  if (target != NULL) {
556  target->call(args, force, debug, dummy);
557  // get lib name
558  auto out_path = target->gen_out_name();
559  unsigned int end_slash = out_path.find_last_of('/');
560  unsigned int end_dot = out_path.find_last_of('.');
561  std::string out =
562  out_path.substr(end_slash + 4, end_dot - end_slash - 4);
563  // Add lib include
564  this->add_library_include(out);
565  // Add all needed for linking to that library
566  this->add_library_dir(".", CBUILD_BUILD_DIR + "/" +
567  target->get_id() + "/" +
568  CBUILD_BUILD_OUT_DIR + "/");
569  this->add_compile_arg("-Wl,-rpath,\"\\$ORIGIN/../../" +
570  target->get_id() + "/" +
571  CBUILD_BUILD_OUT_DIR + "\"");
572  this->add_link_arg("-Wl,-rpath,\"\\$ORIGIN/../../" +
573  target->get_id() + "/" + CBUILD_BUILD_OUT_DIR +
574  "\"");
575  }
576  }
577  // 4 steps
578  if (!force)
579  CBuild::print("Using precompiled object were possible ",
580  CBuild::color::MAGENTA);
581  CBuild::print("Running pre build tasks ", CBuild::GREEN);
582  this->pre_build();
583  CBuild::print("Running build tasks ", CBuild::GREEN);
584  CBuild::print("Now you can see compiler output", CBuild::MAGENTA);
585  this->build();
586  CBuild::print("Running post build tasks ", CBuild::GREEN);
587  this->post_build();
588  CBuild::print("Running pre link tasks ", CBuild::GREEN);
589  this->pre_link();
590  CBuild::print("Running link tasks ", CBuild::GREEN);
591  CBuild::print("Now you can see linker output", CBuild::MAGENTA);
592  // 5-th step - special linking for static libraries
593  if (this->build_type == CBuild::STATIC_LIBRARY) {
594  this->link_pack();
595  } else {
596  this->link();
597  }
598  // Last, 6-th step
599  CBuild::print("Running post link tasks ", CBuild::GREEN);
600  this->post_link();
601  CBuild::print("Calling tasks marked as POST ", CBuild::GREEN);
602  // Call all dependencies (tasks) marked as POSt
603  for (unsigned int i = 0; i < this->required.size(); i++) {
604  auto elem = this->required.at(i);
605  if (elem.data == CBuild::POST) {
606  CBuild::Registry::CallTask(elem.key, {});
607  }
608  }
609  CBuild::print("End of execution of toolchain " + this->id + " ",
610  CBuild::RED);
611 }
