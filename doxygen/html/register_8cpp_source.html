182 void CBuild::Registry::RegisterTask(CBuild::Task *task) {
183  // Save task ptr connected to it's name
184  try {
185  Registry::tasks.push_back_check(task->self_name(), task);
186  } catch (std::exception &e) {
187  CBuild::print_full("Error: trying to register task with the same id: " +
188  task->self_name(),
189  CBuild::RED);
190  return;
191  }
192  // Save task excution state connected to it's name
193  Registry::task_executed.push_back(task->self_name(), false);
194 }
195 void CBuild::Registry::CallTask(std::string name,
196  std::vector<std::string> args) {
197  // Load executed state of task
198  auto check = Registry::task_executed.get_ptr(name);
199  // Check if task exixtst and wasn't executed
200  if (check != NULL && check->data == false) {
201  // Load task
202  auto mod = Registry::tasks.get_ptr(name);
203  // Call all required tasks, recursion ;)
204  for (std::string s : mod->data->self_required()) {
205  Registry::CallTask(s, args);
206  }
207  // Call task
208  mod->data->call(args);
209  // Mark this task es exuted, very easy because of pointers ;)
210  check->data = true;
211  }
212  // Print error if task not found
213  if (check == NULL)
214  CBuild::print("Task " + name + " not found. Exiting ...", CBuild::RED);
215 }
