582 std::string CBuild::get_random_string(unsigned int length) {
583  // Random string and uniform distribution, for generating char
584  const std::string characters =
585  "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz_-";
586  std::uniform_int_distribution<> dis(0, characters.size() - 1);
587  // Random generator
588  std::mt19937_64 gen;
589 #ifdef __linux__
590  // Open random device
591  std::ifstream urandom("/dev/urandom", std::ios::binary);
592  if (!urandom) {
593  goto pseudo;
594  }
595 
596  // Get random data from /dev/urandom
597  std::random_device::result_type seed;
598  urandom.read(reinterpret_cast<char *>(&seed), sizeof(seed));
599 
600  // Seed random generator
601  {
602  CBuild::print_full("Using /dev/urandom for seed", CBuild::MAGENTA);
603  std::seed_seq seedSeq{seed};
604  gen.seed(seedSeq);
605  goto gen;
606  }
607 pseudo:
608 #endif
609  // Seed random generator - fallback
610  {
611  CBuild::print_full("Using std::random_device for seed",
612  CBuild::MAGENTA);
613  std::random_device rd;
614  gen.seed(rd());
615  }
616 gen:
617  // Generate random string with specified length
618  std::string randomString;
619  for (unsigned int i = 0; i < length; ++i) {
620  randomString += characters[dis(gen)];
621  }
622  return randomString;
623 }
