38void CBuild::makefile_out::generate(CBuild::RType mode,
39 lib::map<std::string, std::string> *args) {
40 CBuild::print("Generating Makefile ...", CBuild::color::MAGENTA);
41 // Get command log
42 auto log = CBuild::get_log();
43 // Create Makefile if needed
44 if (!CBuild::fs::exists(this->OUT)) {
45 CBuild::fs::create({this->OUT}, CBuild::fs::FILE);
46 }
47 // Open file
48 std::ofstream makefile;
49 makefile.open(this->OUT, std::ios::app);
50 std::string name = "";
51 // Create name of make task, first part - CBuild run type or
52 // cbuild_ placeholder
53 switch (mode) {
54 case CBuild::BUILD:
55 name += "build_";
56 break;
57 case CBuild::BUILD_RUN:
58 name += "build_run_";
59 break;
60 case CBuild::RUN:
61 name += "run_";
62 break;
63 case CBuild::CLEAR:
64 name += "clear_";
65 break;
66 case CBuild::DEBUG:
67 name += "debug_";
68 break;
69 case CBuild::TASK:
70 name += "task_";
71 break;
72 default:
73 name += "cbuild_";
74 break;
75 }
76 // Second part of make task - target or task id or cbuild
77 // placeholder
78 if (mode == CBuild::BUILD || mode == CBuild::BUILD_RUN ||
79 mode == CBuild::RUN || mode == CBuild::DEBUG || mode == CBuild::CLEAR) {
80 name += *(args->get("toolchain_id"));
81 } else if (mode == CBuild::TASK) {
82 name += *(args->get("task_id"));
83 } else {
84 name += CBuild::get_random_string(6);
85 }
86 // Write all executed command to Makefile
87 makefile << "\n";
88 makefile << name << ":\n";
89 for (auto elem : *log) {
90 makefile << "\t" << elem << "\n";
91 }
92 // End Makefile
93 makefile << "\n";
94 makefile.close();
95}
