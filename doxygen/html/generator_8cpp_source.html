41void CBuild::makefile_out::generate(CBuild::RType mode,
42 lib::map<std::string, std::string> *args) {
43 CBuild::print("Generating Makefile ...", CBuild::color::MAGENTA);
44 // Get command log
45 auto log = CBuild::get_log();
46 // Create Makefile if needed
47 if (!CBuild::fs::exists(this->OUT)) {
48 CBuild::fs::create({this->OUT}, CBuild::fs::FILE);
49 }
50 // Open file
51 std::ofstream makefile;
52 makefile.open(this->OUT, std::ios::app);
53 std::string name = "";
54 // Create name of make task, first part - CBuild run type or
55 // cbuild_ placeholder
56 switch (mode) {
57 case CBuild::BUILD:
58 name += "build_";
59 break;
60 case CBuild::BUILD_RUN:
61 name += "build_run_";
62 break;
63 case CBuild::RUN:
64 name += "run_";
65 break;
66 case CBuild::CLEAR:
67 name += "clear_";
68 break;
69 case CBuild::DEBUG:
70 name += "debug_";
71 break;
72 case CBuild::TASK:
73 name += "task_";
74 break;
75 default:
76 name += "cbuild_";
77 break;
78 }
79 // Second part of make task - target or task id or cbuild
80 // placeholder
81 if (mode == CBuild::BUILD || mode == CBuild::BUILD_RUN ||
82 mode == CBuild::RUN || mode == CBuild::DEBUG || mode == CBuild::CLEAR) {
83 name += *(args->get("toolchain_id"));
84 } else if (mode == CBuild::TASK) {
85 name += *(args->get("task_id"));
86 } else {
87 name += CBuild::get_random_string(6);
88 }
89 // Write all executed command to Makefile
90 makefile << "\n";
91 makefile << name << ":\n";
92 for (auto elem : *log) {
93 makefile << "\t" << elem << "\n";
94 }
95 // End Makefile
96 makefile << "\n";
97 makefile.close();
98}

190void CBuild::ccj_out::generate(CBuild::RType mode,
191 lib::map<std::string, std::string> *args) {
192 // Fail fast if we not building
193 if (mode != CBuild::BUILD) {
194 CBuild::print("Not in build mode, ERROR!", CBuild::RED);
195 return;
196 }
197 // Cannot generate compile_commands.json for meta-toolchain 'all'
198 if (*(args->get("toolchain_id")) == std::string("all")) {
199 CBuild::print(
200 "Cannot generate compile commands for meta-toolchain \"all\"",
201 CBuild::RED);
202 return;
203 }
204 // Get commands
205 // Vars
206 lib::map<std::string, CBuild::cmd> data;
207 // Get data from target
208 auto toolchain =
209 CBuild::Registry::GetToolchain(*(args->get("toolchain_id")));
210 std::vector<std::string> dummy_vec = {};
211 toolchain->call(&dummy_vec, true, false, true);
212 auto files = toolchain->gen_file_list_force();
213 auto cmds = toolchain->get_cmds();
214 auto targs = toolchain->get_compile_args();
215 auto dir = args->get_ptr("curr_path")->data;
216 for (auto file : files) {
217 data.push_back({file.key,
218 {.base_path = dir,
219 .cmd = CBuild::eval_cmd(std::array<std::string, 4>{
220 file.key, file.data, cmds[0], targs}),
221 .file = file.key,
222 .in_file = false}});
223 }
224 // Store in file
225 CBuild::print("Generating compile_commands.json ...",
226 CBuild::color::MAGENTA);
227 CBuild::print("This can breaks if compile_commands.json already exists and "
228 "this file was not created by CBuild!",
229 CBuild::color::RED);
230 if (CBuild::fs::exists(this->OUT) == false) {
231 CBuild::print_full("File not found, creating file...");
232 CBuild::fs::create({this->OUT}, CBuild::fs::FILE);
233 std::fstream ccj(this->OUT);
234 ccj << "[\n";
235 for (unsigned int j = 0; j < data.size(); j++) {
236 auto elem = data.at(j);
237 if (j == 0) {
238 ccj << "{";
239 } else {
240 ccj << ",\n{\n";
241 }
242 ccj << (std::string("\t\"directory\": \"") + dir +
243 std::string("\",\n"));
244 ccj << (std::string("\t\"command\": \"") +
245 CBuild::preprocess_json_str(elem.data.cmd) +
246 std::string("\",\n"));
247 ccj << (std::string("\t\"file\": \"") + elem.data.file +
248 std::string("\"\n"));
249 ccj << "}";
250 }
251 ccj << "\n]";
252 } else {
253 CBuild::print_full("File found, appending...");
254 CBuild::line_filebuff ccj(this->OUT);
255 ccj.update();
256 int i = 0;
257 try {
258 while (true) {
259 std::string str = ccj.get_line(i);
260 size_t pos = str.find("\"file\":");
261 if (pos != std::string::npos) {
262 // We find file
263 size_t st = str.find_first_of('"', pos + 7);
264 size_t ed = str.find_last_of('"');
265 std::string fname = str.substr(st + 1, ed - st - 1);
266 auto elem = data.get_ptr(fname);
267 if (elem != NULL) {
268 // File is recompiled, so replace "command"
269 CBuild::print_full(
270 std::string("Found file \"") + fname +
271 std::string("\" that is recompiled, replacing "
272 "\"command\" entry"));
273 ccj.del_line(i - 1);
274 ccj.set_line(
275 std::string("\t\"command\": \"") +
276 CBuild::preprocess_json_str(elem->data.cmd) +
277 std::string("\","),
278 i - 1);
279 // Mark file as processed
280 elem->data.in_file = true;
281 }
282 }
283 i++;
284 }
285 } catch (std::exception &e) {
286 // Add coma at the end, no worry about this later
287 ccj.del_line(i - 2);
288 ccj.set_line("},", i - 2);
289 // Check if all files are processed
290 for (unsigned int j = 0; j < data.size(); j++) {
291 auto elem = data.at(j);
292 if (elem.data.in_file == false) {
293 CBuild::print_full(std::string("Found new file: \"") +
294 elem.data.file +
295 std::string("\", inserting new entry"));
296 // If file is not in database - insert entry, we dont need
297 // to sert in_file tag, because this is last part, where
298 // data map is valid
299 ccj.set_line("{", i - 1);
300 i++;
301 ccj.set_line(std::string("\t\"directory\": \"") + dir +
302 std::string("\","),
303 i - 1);
304 i++;
305 ccj.set_line(
306 std::string("\t\"command\": \"") +
307 CBuild::preprocess_json_str(elem.data.cmd) +
308 std::string("\","),
309 i - 1);
310 i++;
311 ccj.set_line(std::string("\t\"file\": \"") +
312 elem.data.file + std::string("\""),
313 i - 1);
314 i++;
315 ccj.set_line("},", i - 1);
316 i++;
317 }
318 }
319 // Remove comma after last element as per json specs
320 ccj.del_line(i - 2);
321 ccj.set_line("}", i - 2);
322 ccj.update();
323 }
324 }
