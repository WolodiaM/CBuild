145CBuild::types::file_content get_file_data(std::string path,
146 std::string included_from = "") {
147 // Return variable
148 std::string p = path;
149 // Set path to file
150 try {
151 p = CBuild::fs::normalize_path(path, included_from);
152 } catch (const std::exception &e) {
153 CBuild::print_full(std::string("Error in path creation, path - \"") + path +
154 std::string("\", included_from - \"") +
155 included_from + std::string("\""),
156 CBuild::RED);
157 exit(0xFF);
158 }
159 CBuild::types::file_content f;
160 f.path = p;
161 // Load file content to string
162 std::ifstream file(path);
163 std::stringstream buff;
164 buff << file.rdbuf();
165 file.close();
166 std::string str = buff.str();
167 // Strip comments from file
168 str = std::regex_replace(str, CBuild::consts::cxx_line_comment, "");
169 str = std::regex_replace(str, CBuild::consts::c_multiline_comment, "");
170 str = std::regex_replace(str, CBuild::consts::doxygen_comment, "");
171 f.content = str;
172 // Get includes
173 std::smatch match;
174 std::string::const_iterator start(str.cbegin());
175 // Base file path
176 std::string base = CBuild::fs::base(path);
177 while (std::regex_search(start, str.cend(), match,
178 CBuild::consts::include_parser)) {
179 // Get file
180 std::string head = match[1];
181 bool err = false;
182 // Convert to absolute path and check for system includes
183 try {
184 head = CBuild::fs::normalize_path(head, base);
185 } catch (const std::exception &e) {
186 err = true;
187 }
188 // Push path, if it is project include
189 if (!err) {
190 f.includes.push_back(head);
191 }
192 start = match.suffix().first;
193 }
194 return f;
195}

199// * @param str => std::string -> Input string
200// * @return uint64_t -> Hash of data

202// uint64_t hash(std::string str) {

215uint64_t hash(std::string str) {
216 uint64_t hash = 14695981039346656037ULL;
217 for (char c : str) {
218 hash ^= static_cast<uint64_t>(c);
219 hash *= 1099511628211ULL;
220 }
221 return hash;
222}
230void procces_files(std::vector<std::string> files,
231 std::string toolchain_id __attribute_maybe_unused__) {
232 CBuild::print_full("\t\t\tCBuild hash v3.0", CBuild::color::MAGENTA);
233 // For every input file
234 for (auto file : files) {
235 // Get file data
236 auto data = CBuild::get_file_data(file);
237 auto cpp = CBuild::types::file(data.path);
238 // Get file hash
239 cpp.hash_new = CBuild::hash(data.content);
240 CBuild::print_full("Calculating hash for \"" + cpp.abs_path + "\"",
241 CBuild::color::GREEN);
242 CBuild::print_full("Need to calculate hash...", CBuild::color::RED);
243 auto old = CBuild::vars::old_hashes.get(cpp.abs_path);
244 if (old != NULL) {
245 cpp.hash_old.set(*old);
246 }
247 // For every invclude
248 for (auto inc : data.includes) {
249 // Get file data
250 auto hdata = CBuild::get_file_data(inc, cpp.abs_path);
251
252 auto hpp = CBuild::types::file(hdata.path);
253 // Check if hash is already stored in hash
254 const uint64_t *hash = NULL;
255 CBuild::print_full("Calculating hash for \"" + hpp.abs_path + "\"",
256 CBuild::color::GREEN);
257 if ((hash = CBuild::vars::headers.get(hpp.abs_path)) != NULL) {
258 // Use stored hash
259 hpp.hash_new = *hash;
260 CBuild::print_full("Using hash from cache.", CBuild::color::GREEN);
261 } else {
262 // Calculate file hash
263 hpp.hash_new = CBuild::hash(hdata.content);
264 CBuild::print_full("Need to calculate hash...", CBuild::color::RED);
265 }
266 auto old = CBuild::vars::old_hashes.get(hpp.abs_path);
267 if (old != NULL) {
268 hpp.hash_old.set(*old);
269 }
270 // Store hash
271 try {
272 CBuild::vars::headers.push_back_check(hpp.abs_path, hpp.hash_new);
273 } catch (std::exception &e) {
274 // So, we already have this hash, we dont need to do anything
275 }
276 // Store proccesed include
277 cpp.includes.push_back(hpp);
278 }
279 // Store proccesed file
280 CBuild::vars::filelist.push_back(cpp);
281 }

305void load_hash(std::string toolchain_id) {
306 std::string path = CBUILD_BUILD_DIR + "/" + toolchain_id + "/" +
307 CBUILD_HASH_DIR + "/" + CBUILD_HASH_FILE;
308 if (!CBuild::fs::exists(path)) {
309 return;
310 }
311 std::ifstream file(path);
312 std::string line;
313 while (std::getline(file, line)) {
314 std::istringstream iss(line);
315 std::string key;
316 uint64_t value;
317 iss >> key >> std::hex >> value;
318
319 try {
320 CBuild::vars::old_hashes.push_back_check(key, value);
321 } catch (const std::exception &e) {
322 // Maybe we have some previvous errors, and somehowe we have two
323 // simmilar hashes, so, save only first, in future we fix this by
324 // fully recreating file
325 }
326 }
327 file.close();
