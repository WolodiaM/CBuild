52ARG_TYPE out_handler(lib::map<std::string, std::string> *args, char **argv,

131 CBuild::exit(0xFF);

133 // Now we need to figure why user run this app
134 // Lets save current token to std::string to ease of use
135 std::string token = argv[ptr];
136 // There go boring comparisions
137 if (token == std::string("-b")) {
138 *type = CBuild::BUILD;
139 } else if (token == std::string("-br")) {
140 *type = CBuild::BUILD_RUN;
141 } else if (token == std::string("-r")) {
142 *type = CBuild::RUN;
143 } else if (token == std::string("-d")) {
144 *type = CBuild::DEBUG;
145 } else if (token == std::string("-c")) {
146 *type = CBuild::CLEAR;
147 } else if (token == std::string("-ld")) {
148 *type = CBuild::LOAD_DEPS;
149 }
150 // Get toolchains and check if one of them match our needed
151 std::vector<std::string> tools = CBuild::Registry::GetToolchainsList();
152 bool err = true;
153 for (std::string tool : tools) {
154 if (tool == std::string(argv[ptr + 1])) {
155 err = false;
156 }
157 }
158 // We have exception for -b all
159 if (*type == CBuild::BUILD &&
160 std::string(argv[ptr + 1]) == std::string("all")) {
161 err = false;
162 }
163 // If no match, yes, we are safe to crash here and remaind proper format of
164 // the argument to user
165 if (err == true) {
166 CBuild::print(std::string("Illegal or unregistered toolchain name! - \"") +

168 CBuild::RED);
169 CBuild::print(std::string("Usage: ") + std::string(argv[ptr]) +

171 CBuild::GREEN);
172 CBuild::exit(0xFF);

291CBuild::RType CBuild::parse(lib::map<std::string, std::string> *args, int argc,
292 char **argv) {
293 // Init CBuild registry
294 CBuild::Registry::init();
295 // Some internal variables
296 CBuild::RType ret = CBuild::ERROR;
297 int ptr = 1;
298 auto keywords = CBuild::Registry::GetKeywordsList();
299 // For every token
300 while (ptr < argc) {
301 // Get handler
302 auto h = CBuild::parsers_list.get_ptr(argv[ptr]);
303 CBuild::ARG_TYPE inc = CBuild::BLANK_ARG;
304 // If it is available
305 if (h != NULL) {
306 // Call it
307 inc = h->data(args, argv, argc, ptr, &ret);
308 } else {
309 // Else check for custom keywords
310 const std::string *task = keywords.get(argv[ptr]);
311 if (task != NULL) {
312 ret = CBuild::TASK;
313 try {
314 args->push_back_check("task_id", *task);
315 } catch (std::exception &e) {
316 }
317 inc = CBuild::ARG_TYPE::SIMPLE_ARG;
318 } else {
319 // Else - error
320 CBuild::print(std::string("Illegal argument! - \"") +
321 std::string(argv[ptr]) + std::string("\""),
322 CBuild::RED);
323 exit(0xFF);
324 }
325 }
326 // Handler must return number of consumed tokens, so, increment pointer
327 // by this number
328 ptr += inc;
329 }
330 return ret;
331}
