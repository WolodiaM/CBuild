318CBuild::RType CBuild::parse(lib::map<std::string, std::string> *args, int argc,
319 char **argv) {
320 // Init CBuild registry
321 CBuild::Registry::init();
322 // Some internal variables
323 CBuild::RType ret = CBuild::ERROR;
324 int ptr = 1;
325 auto keywords = CBuild::Registry::GetKeywordsList();
326 // For every token
327 while (ptr < argc) {
328 // Get handler
329 auto h = CBuild::arguments.get_ptr(argv[ptr]);
330 CBuild::ARG_TYPE inc = CBuild::BLANK_ARG;
331 // If it is available
332 if (h != NULL) {
333 // Call it
334 inc = h->data(args, argv, argc, ptr, &ret);
335 } else {
336 // Else check for custom keywords
337 const std::string *task = keywords.get(argv[ptr]);
338 if (task != NULL) {
339 ret = CBuild::TASK;
340 try {
341 args->push_back_check("task_id", *task);
342 } catch (std::exception &e) {
343 }
344 inc = CBuild::ARG_TYPE::SIMPLE_ARG;
345 } else {
346 // Else - error
347 CBuild::print(std::string("Illegal argument! - \"") +
348 std::string(argv[ptr]) + std::string("\""),
349 CBuild::RED);
350 exit(0xFF);
351 }
352 }
353 // Handler must return number of consumed tokens, so, increment pointer
354 // by this number
355 ptr += inc;
356 }
357 return ret;
358}
