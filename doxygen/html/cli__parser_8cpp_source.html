30/* SPECIFICATIONS
31 * - Parse arguments one by one
32 * - If invelid argument/argument with modifier, not have modifier - exit with
33 * error message
34 * - If no args - exit with error message
35 * - If one of internl tasks (--init, --version, --help) - clear all and return
36 *immideately
37 * - map entries:
38 * a<num> - argumenent, val = string
39 * pa<num> - program argument, val = string
40 * force - force mode, val = "force"
41 * out - output mode, val = "verbose", "none"
42 * gen - generator, val = CBuild::Registry::GetGeneratorsList()
43 * toolchain_id - toolchain, val = CBuild::Registry::GetToolchainsList()
44 * task_id - task, val = CBuild::Registry::GetTasksList(), "CBuild_init",
45 * "CBuild_version", "CBuild_help"
46 *
47 *## CLI arguments
48 * `-f` - force recompilation of all files.
49 * `-v` - verbose output, some addition info and some debugging info, mask
50 * `--no-cli-out` if used before.
51 * `--no-cli-out` - no output from CBuild at all, mask `-v` if used before.
52 * `-g make` - generate Makefile based on output from CBuild, simly log all
53 * executed command and pack it in Makefile syntax, so, sadly, CBuild::fs
54 * calls not propagate to Makefile.
55 * `-g ccj` - generate compile_commands.json, use internal command evaluator in
56 * CBuild::Toolchain, from the box work only for toolchains, that use
57 * standard compiler calling convention, see below.
58 * `-ld <toolchain_name>` - load dependencies of specified target, only for
59 * CBuild dependencies.
60 * `-b <toolchain_name>` - build specified target.
61 * `-b all` - build all registered targets
62 * `-r <toolchain_name>` - run specified target.
63 * `-br <toolchain_name>` - build and then run specified target.
64 * `-d <toolchain_name>` - build and then run specified target in debug mode.
65 * Now work only with gdb on linux.
66 * `-c <toolchain_name>` - clear all data of specified target.
67 * `-t <task_name>` - run specified task.
68 * `-a <arg>` - set some argument (see it as #define arg, without value). It can
69 * be used by ran toolchain or ran task. No preprocessing, simple
70 * propagation for user-defined arguments, preserver order in what
71 * arguments is specified.
72 * `-pa <arg>` - specify argument for ran app, propagated to app ran by `-r` and
73 * `-br`, in `-d` arguments propagates to debugger.
74 * `--rebuild` - rebuild CBuild executable.
75 * `--init` - init CBuild folder structure, not always required.
76 * `--help` - print help message - this information.
77 * `--version` - print curent version of CBuild
78 * `--get-cli-autocompetition` - generate autocompetition data for bash, usage
79 * with custom autocompetition script
80 */

160ARG_TYPE build_handler(lib::map<std::string, std::string> *args, char **argv,
161 int argc, int ptr, CBuild::RType *type) {
162 // User do not provide needed argument
163 if (ptr >= (argc + 1)) {
164 CBuild::print("Illegal command format!", CBuild::RED);
165 CBuild::print(std::string("Usage: ") + std::string(argv[ptr]) +
166 std::string(" <toolchain id>"),
167 CBuild::GREEN);
168 exit(0xFF);
169 }
170 // Now we need to figure why user run this app
171 // Lets save current token to std::string to ease of use
172 std::string token = argv[ptr];
173 // There go boring comparisions
174 if (token == std::string("-b")) {
175 *type = CBuild::BUILD;
176 } else if (token == std::string("-br")) {
177 *type = CBuild::BUILD_RUN;
178 } else if (token == std::string("-r")) {
179 *type = CBuild::RUN;
180 } else if (token == std::string("-d")) {
181 *type = CBuild::DEBUG;
182 } else if (token == std::string("-c")) {
183 *type = CBuild::CLEAR;
184 } else if (token == std::string("-ld")) {
185 *type = CBuild::RType::LOAD_DEPS;
186 }
187 // Get toolchains and check if one of them match our needed
188 std::vector<std::string> tools = CBuild::Registry::GetToolchainsList();
189 bool err = true;
190 for (std::string tool : tools) {
191 if (tool == std::string(argv[ptr + 1])) {
192 err = false;
193 }
194 }
195 // We have exception for -b all
196 if (*type == CBuild::BUILD &&
197 std::string(argv[ptr + 1]) == std::string("all")) {
198 err = false;
199 }
200 // If no match, yes, we are safe to crash here and remaind proper format of
201 // the argument to user
202 if (err == true) {
203 CBuild::print(std::string("Usage: ") + std::string(argv[ptr]) +
204 std::string(" <toolchain id>"),
205 CBuild::GREEN);
206 exit(0xFF);
207 }
208 // If all good - save selected toolchain for later
209 args->push_back_check("toolchain_id", argv[ptr + 1]);
210 return CBuild::ARG_TYPE::TOOLCHAIN_ARG;

223CBuild::RType CBuild::parse(lib::map<std::string, std::string> *args, int argc,
224 char **argv) {
225 // Init CBuild registry
226 CBuild::Registry::init();
227 // Some internal variables
228 CBuild::RType ret = CBuild::ERROR;
229 int ptr = 1;
230 // For every token
231 while (ptr <= argc) {
232 // Get handler
233 auto h = CBuild::arguments.get_ptr(argv[ptr]);
234 CBuild::ARG_TYPE inc = CBuild::BLANK_ARG;
235 // If it is available
236 if (h != NULL) {
237 // Call it
238 inc = h->data(args, argv, argc, ptr, &ret);
239 } else {
240 // Else - error
241 CBuild::print("Illegal argument");
242 exit(0xFF);
243 }
244 // Handler must return number of cinsumed tokes, so, incre,ent pointer by
245 // this number
246 ptr += inc;
247 }
248 return ret;
249}
