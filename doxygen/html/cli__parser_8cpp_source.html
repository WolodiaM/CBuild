117ARG_TYPE build_handler(lib::map<std::string, std::string>* args, char** argv, int argc, int ptr,
118 CBuild::RType* type) {
119 // User do not provide needed argument (ptr is current argument, so (ptr +
120 // 1) is next argument
121 if (argc <= (ptr + 1)) {
122 CBuild::print("Illegal command format!", CBuild::RED);
123 CBuild::print(std::string("Usage: ") + std::string(argv[ptr]) +
124 std::string(" <toolchain id>"),
125 CBuild::GREEN);
126 CBuild::exit(0xFF);
127 }
128 // Now we need to figure why user run this app
129 // Lets save current token to std::string to ease of use
130 std::string token = argv[ptr];
131 // There go boring comparisions
132 if (token == std::string("-b")) {
133 *type = CBuild::BUILD;
134 } else if (token == std::string("-br")) {
135 *type = CBuild::BUILD_RUN;
136 } else if (token == std::string("-r")) {
137 *type = CBuild::RUN;
138 } else if (token == std::string("-d")) {
139 *type = CBuild::DEBUG;
140 } else if (token == std::string("-c")) {
141 *type = CBuild::CLEAR;
142 } else if (token == std::string("-ld")) {
143 *type = CBuild::LOAD_DEPS;
144 }
145 // Get toolchains and check if one of them match our needed
146 std::vector<std::string> tools = CBuild::Registry::GetToolchainsList();
147 bool err = true;
148 for (std::string tool : tools) {
149 if (tool == std::string(argv[ptr + 1])) {
150 err = false;
151 }
152 }
153 // We have exception for -b all
154 if (*type == CBuild::BUILD && std::string(argv[ptr + 1]) == std::string("all")) {
155 err = false;
156 }
157 // If no match, yes, we are safe to crash here and remaind proper format of
158 // the argument to user
159 if (err == true) {
160 CBuild::print(std::string("Illegal or unregistered toolchain name! - \"") +
161 std::string(argv[ptr + 1]) + std::string("\""),
162 CBuild::RED);
163 CBuild::print(std::string("Usage: ") + std::string(argv[ptr]) +
164 std::string(" <toolchain id>"),
165 CBuild::GREEN);
166 CBuild::exit(0xFF);
167 }
168 // If all good - save selected toolchain for later
169 try {
170 args->push_back_check("toolchain_id", argv[ptr + 1]);
171 } catch (std::exception& e) {
172 CBuild::print_full(
173 std::string("Toolchain is already set! New toolchain - \"") +
174 std::string(argv[ptr + 1]) + std::string("\", old toolchain is - \"") +
175 std::string(args->get_ptr("toolchain_id")->data) + std::string("\"!"),
176 CBuild::RED);
177 CBuild::exit(0xFF);
178 }
179 return CBuild::ARG_TYPE::TOOLCHAIN_ARG;

279CBuild::RType CBuild::parse(lib::map<std::string, std::string>* args, int argc, char** argv) {
280 // Init CBuild registry
281 CBuild::Registry::init();
282 // Some internal variables
283 CBuild::RType ret = CBuild::ERROR;
284 int ptr = 1;
285 auto keywords = CBuild::Registry::GetKeywordsList();
286 // For every token
287 while (ptr < argc) {
288 // Get handler
289 auto h = CBuild::parsers_list.get_ptr(argv[ptr]);
290 CBuild::ARG_TYPE inc = CBuild::BLANK_ARG;
291 // If it is available
292 if (h != NULL) {
293 // Call it
294 inc = h->data(args, argv, argc, ptr, &ret);
295 } else {
296 // Else check for custom keywords
297 const std::string* task = keywords.get(argv[ptr]);
298 if (task != NULL) {
299 ret = CBuild::TASK;
300 try {
301 args->push_back_check("task_id", *task);
302 } catch (std::exception& e) {
303 }
304 inc = CBuild::ARG_TYPE::SIMPLE_ARG;
305 } else {
306 // Else - error
307 CBuild::print(std::string("Illegal argument! - \"") + std::string(argv[ptr]) +
308 std::string("\""),
309 CBuild::RED);
310 exit(0xFF);
311 }
312 }
313 // Handler must return number of consumed tokens, so, increment pointer
314 // by this number
315 ptr += inc;
316 }
317 return ret;
318}
