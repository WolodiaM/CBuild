291 CBuild::RType CBuild::parse(lib::map<std::string, std::string> *args, int argc,
292  char **argv) {
293  // Init CBuild registry
294  CBuild::Registry::init();
295  // Some internal variables
296  CBuild::RType ret = CBuild::ERROR;
297  int ptr = 1;
298  auto keywords = CBuild::Registry::GetKeywordsList();
299  // For every token
300  while (ptr < argc) {
301  // Get handler
302  auto h = CBuild::parsers_list.get_ptr(argv[ptr]);
303  CBuild::ARG_TYPE inc = CBuild::BLANK_ARG;
304  // If it is available
305  if (h != NULL) {
306  // Call it
307  inc = h->data(args, argv, argc, ptr, &ret);
308  } else {
309  // Else check for custom keywords
310  const std::string *task = keywords.get(argv[ptr]);
311  if (task != NULL) {
312  ret = CBuild::TASK;
313  try {
314  args->push_back_check("task_id", *task);
315  } catch (std::exception &e) {
316  }
317  inc = CBuild::ARG_TYPE::SIMPLE_ARG;
318  } else {
319  // Else - error
320  CBuild::print(std::string("Illegal argument! - \"") +
321  std::string(argv[ptr]) + std::string("\""),
322  CBuild::RED);
323  exit(0xFF);
324  }
325  }
326  // Handler must return number of consumed tokens, so, increment pointer
327  // by this number
328  ptr += inc;
329  }
330  return ret;
331 }
