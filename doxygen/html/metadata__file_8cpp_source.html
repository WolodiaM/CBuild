32int read_file_metadata_internal(std::string path, CBuild::source_metadata_file* metadata) {
33 if (!CBuild::fs::exists(path)) {
34 metadata = NULL;
35 return -1;
36 }
37 std::fstream f;
38 std::string FILE_DATA;
39 f.open(path);
40 std::getline(f, FILE_DATA, '\0');
41 f.close();
42 std::string line;
43 std::stringstream FILE_DATA_STREAM(FILE_DATA);
44 std::getline(FILE_DATA_STREAM, line, '\n');
45 metadata->source = line;
46 std::getline(FILE_DATA_STREAM, line, '\n');
47 metadata->object = line;
48 std::getline(FILE_DATA_STREAM, line, '\n');
49 if (line != std::string("DEPS")) {
50 return -3;
51 }
52 std::getline(FILE_DATA_STREAM, line, '\n');
53 while (line != std::string("ENDDEPS")) {
54 metadata->deps.push_back(line);
55 std::getline(FILE_DATA_STREAM, line, '\n');
56 }
57 std::getline(FILE_DATA_STREAM, line, '\n');
58 if (!std::isdigit(line.front()) || !std::isdigit(line.back())) {
59 return -3;
60 }
61 metadata->hash = std::strtoull(line.c_str(), NULL, 10);
62 return 0;
63}

166std::string CBuild::get_file_metadata_path(std::string target_id, std::string file) {
167 // Replace '/' with '.'
