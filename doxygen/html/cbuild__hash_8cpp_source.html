77 void parse_file(internal_data* data, std::string target_id, std::string file, std::string object) {
78  extended_metadata m;
79  // Get old metadata
80  int err = CBuild::read_file_metadata(target_id, file, &(m.old_meta));
81  m.old_meta_available = (err == 0);
82  // Get new metadata: file and object
83  m.new_meta.source = file;
84  m.new_meta.object = object;
85  // Read file content
86  std::ifstream f(file);
87  std::string file_content;
88  std::getline(f, file_content, '\0');
89  // Strip comments from file
90  file_content = std::regex_replace(file_content, cpp_comment, "");
91  file_content = std::regex_replace(file_content, c_comment, "");
92  // Get new metadata: hash
93  m.new_meta.hash = CBuild::hash_fnv_1a(file_content);
94  // Get new metadata: get dependencies and add new headers for parsing
95  m.new_meta.deps = {};
96  std::smatch match;
97  std::string::const_iterator start(file_content.cbegin());
98  while (std::regex_search(start, file_content.cend(), match, include_detector)) {
99  std::string tmp = "";
100  try {
101  tmp = CBuild::fs::normalize_relative_path(match[1], CBuild::fs::base(file));
102  } catch (std::exception& e) {
103  }
104  if (tmp != std::string("") && CBuild::fs::exists(tmp)) {
105  m.new_meta.deps.push_back(tmp);
106  try {
107  data->headers_to_parse.push_back_check(tmp, false);
108  } catch (std::exception& e) {
109  }
110  }
111  start = match.suffix().first;
112  }
113  // Store metadata into a shared data
114  if (object == std::string("header")) {
115  try {
116  data->headers_metadata.push_back_check(file, m);
117  } catch (std::exception& e) {
118  }
119  } else {
120  try {
121  data->sources_metadata.push_back_check(file, m);
122  } catch (std::exception& e) {
123  }
124  }
125  // Rewrite metadata with new data
126  CBuild::write_file_metadata(target_id, file, &(m.new_meta));
127 }

150 bool check_mismatch_recursive(const extended_metadata* file, internal_data* data) {
151  // std::printf("Checking file '%s'\n", file->new_meta.source.c_str());
152  if (file->old_meta_available == false) {
153  // std::printf("File: '%s' -> no_old_meta\n", file->new_meta.source.c_str());
154  return true;
155  }
156  if (file->new_meta.object != file->old_meta.object) {
157  // std::printf("File: '%s' -> mismatch object\n", file->new_meta.source.c_str());
158  return true;
159  }
160  if (file->new_meta.hash != file->old_meta.hash) {
161  // std::printf("File: '%s' -> mismatch hash\n", file->new_meta.source.c_str());
162  return true;
163  }
164  if (file->new_meta.deps != file->old_meta.deps) {
165  // std::printf("File: '%s' -> mismatch deps\n", file->new_meta.source.c_str());
166  return true;
167  }
168  for (auto dep : file->new_meta.deps) {
169  auto header = data->headers_metadata.get(dep);
170  if (header != NULL) {
171  if (check_mismatch_recursive(header, data)) {
172  return true;
173  }
174  }
175  auto source = data->sources_metadata.get(dep);
176  if (source != NULL) {
177  if (check_mismatch_recursive(source, data)) {
178  return true;
179  }
180  }
181  }
182  return false;
183 }

208  for (auto elem : data.headers_metadata) {
209  print_metadata(&(elem.data));
210  }
211  lib::map<std::string, std::string> ret;
212  // Get files for recompilation
213  for (auto file : data.sources_metadata) {
214  if (check_mismatch_recursive(&(file.data), &data)) {
215  ret.push_back(file.data.new_meta.source, file.data.new_meta.object);
